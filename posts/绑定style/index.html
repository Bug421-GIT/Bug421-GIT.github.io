绑定style

发表于 2023/10/22
作者 张晓岸 1 分钟阅读
  1. Class样式 —— :class="xxx"
    • 字符串写法：类目不确定，动态获取
    • 对象写法：个数不确定，名字也不确定
    • 数组写法：名字已经确定，个数确定，但不确定用不用
  2. Style样式 —— 属性键值对，需要驼峰命名法
    • :style = "{fontSize: 'xxx'}"
    • :style = "[a,b]"

<div id="root">
    <!-- <div class="base" @click="changeClass" id="demo">test</div> -->
    <!-- 【字符串写法】插入一个class，动态实现class -->
    <div class="base" :class="a" @click="changeClass">test</div>
    <!-- 【数组写法】插入一个系列class，样式个数不确定、名字不确定 -->
    <div class="base" :class="arr">test</div>
    <!--  :class="['radius1','radius2']"  双引号里面可以放 表达式 -->
    <!--  :class="[radius1,radius2]"  去掉数组元素里面的引号，vue当成变量名使用，需要data里面有相应的值 -->

    <!-- 【对象写法】样式个数确定、名字确定，但是动态决定用不用-->
    <div class="base" :class="obj">test</div>
    <!--  :class="{radius1: true,radius2: false}"    -->

    <!-- 【行内式】 1.样式为键值对，需要使用对象  2.font-size 改为fontSize， -->
    <!-- 数值写法：style='stylyArr'  style='[styleObj1,stleObj2]'-->
    <div class="base" :style="{fontSize:fsize+'px'}" >test</div>
    <div class="base" :style="stylyObj" >test</div>
</div>
